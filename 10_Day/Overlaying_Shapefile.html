Overlaying the two Shapefiles with labels¶

# create a plot and set its size
fig, ax = plt.subplots(figsize=(9, 7))

# plot the Ethiopian basins using their basin names, with a colormap and no edgecolor
ET_basins.plot(ax=ax, column='BASINNAME', cmap='tab20c', edgecolor='None', legend=True, alpha=1.0)

# plot the Ethiopian regions with no color and a black edgecolor
ET_regions.plot(ax=ax, color='None', edgecolor='black', legend=True)

# add a grid to the plot
ax.grid(True)

# set the plot's title and font size
ax.set_title('Overlaying Ethiopian basins over the Regions', fontsize=15)

# set the plot's x and y axis labels
ax.set(xlabel='Longitude (Degrees)', ylabel='Latitude (Degrees)')

# add labels to the regions
for idx, row in ET_regions.iterrows():
    plt.annotate(text=row['REGIONNAME'], xy=row['geometry'].centroid.coords[0], ha='center')

# display the plot
plt.show()
